void selection_sort(int arr[], int n) {
    for (int i = n - 1; i > 0; i--) {
        int min_idx = 0;
        for (int j = 1; j <= i; j++) {
            if (arr[j] < arr[min_idx]) {
                min_idx = j;
            }
        }
        int temp = arr[min_idx];
        arr[min_idx] = arr[i];
        arr[i] = temp;
    }
}
